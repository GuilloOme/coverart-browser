00090     def calc_popup_position(self, widget):
00091         # this calculates the popup positioning - algorithm taken
00092         # from Gtk3.8 gtk/gtkmenubutton.c
00093         
00094         toplevel = self.get_toplevel()
00095         toplevel.set_type_hint(Gdk.WindowTypeHint.DROPDOWN_MENU)
00096         
00097         menu_req, pref_req = widget.get_preferred_size()
00098         align = widget.get_halign()
00099         direction = self.get_direction()
00100         window = self.get_window()
00101 
00102         screen = widget.get_screen()
00103         monitor_num = screen.get_monitor_at_window(window)
00104         if (monitor_num < 0):
00105             monitor_num = 0
00106         monitor = screen.get_monitor_workarea(monitor_num)
00107 
00108         allocation = self.get_allocation()
00109 
00110         ret, x,y = window.get_origin()
00111         x += allocation.x
00112         y += allocation.y
00113 
00114         if allocation.width - menu_req.width > 0:
00115             x += allocation.width - menu_req.width
00116 
00117         if ((y + allocation.height + menu_req.height) <= monitor.y + monitor.height):
00118             y += allocation.height
00119         elif ((y - menu_req.height) >= monitor.y):
00120             y -= menu_req.height
00121         else:
00122             y -= menu_req.height
00123         
00124         return x, y

00258     def _getBlendedPixbuf(self, pixbuf):
00259         """Turn a pixbuf into a blended version of the pixbuf by drawing a
00260         transparent alpha blend on it."""
00261         pixbuf = pixbuf.copy()
00262         
00263         w,h = pixbuf.get_width(), pixbuf.get_height()
00264         surface = cairo.ImageSurface(
00265             cairo.FORMAT_ARGB32, pixbuf.get_width(), pixbuf.get_height())
00266         context = cairo.Context(surface)
00267         
00268         Gdk.cairo_set_source_pixbuf(context, pixbuf, 0, 0)
00269         context.paint()
00270 
00271         context.set_source_rgba(32, 32, 32, 0.4)
00272         context.set_line_width(0)
00273         context.rectangle(0, 0, w, h)
00274         context.fill()
00275 
00276         pixbuf = Gdk.pixbuf_get_from_surface(surface, 0, 0, w, h)
00277 
00278         return pixbuf
