00437     @classmethod
00438     def global_filter(cls, searchtext=None):
00439         def filt(album):
00440             # this filter is more complicated: for each word in the search
00441             # text, it tries to find at least one match on the params of
00442             # the album. If no match is given, then the album doesn't match
00443             if not searchtext:
00444                 return True
00445 
00446             words = RB.search_fold(searchtext).split()
00447             params = map(RB.search_fold, [album.name, album.artist,
00448                 album.artists, album.track_titles])
00449             matches = []
00450 
00451             for word in words:
00452                 match = False
00453 
00454                 for param in params:
00455                     if word in param:
00456                         match = True
00457                         break
00458 
00459                 matches.append(match)
00460 
00461             return False not in matches
00462 
00463         return filt

00532     @classmethod

00534         '''
00535         The year is in RATA DIE format so need to extract the year
00536 
00537         The searchdecade param can be None meaning all results
00538         or -1 for all albums older than our standard range which is 1930

00541         def filt(album):
00542             if not searchdecade:
00543                 return True
00544

00551 
00552             if searchdecade > 0:
00553                 return searchdecade == year
00554             else:
00555                 return year < 1930
00556

01268     def search_covers(self, albums=None, callback=lambda *_: None):
01269         '''
01270         Request all the albums' covers, one by one, periodically calling a
01271         callback to inform the status of the process.
01272         The callback should accept one argument: the album which cover is
01273         being requested. When the argument passed is None, it means the
01274         process has finished.
01275 
01276         :param albums: `list` of `Album` for which look for covers.
01277         :param callback: `callable` to periodically inform when an album's
01278             cover is being searched.
01279         '''
01280         if albums is None:
01281             albums = self._album_manager.model.get_all()
01282 
01283         def search_next_cover(*args):
01284             # unpack the data
01285             iterator, callback = args[-1]
01286 
01287             # if the operation was canceled, break the recursion
01288             if self._cancel_cover_request:
01289                 del self._cancel_cover_request
01290                 callback(None)
01291                 return
01292 
01293             #try to obtain the next album
01294             try:
01295                 while True:
01296                     album = iterator.next()
01297 
01298                     if album.cover is self.unknown_cover:
01299                         break
01300 
01301                 # inform we are starting a new search
01302                 callback(album)
01303 
01304                 # request the cover for the next album
01305                 self.search_cover_for_album(album, search_next_cover,
01306                     (iterator, callback))
01307             except StopIteration:
01308                 # inform we finished
01309                 callback(None)
01310             except Exception as e:
01311                 print "Error while searching covers: " + str(e)
01312 
01313         self._cancel_cover_request = False
01314         search_next_cover((iter(albums), callback))
