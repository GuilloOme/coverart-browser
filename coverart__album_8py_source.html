00476     @classmethod
00477     def global_filter(cls, searchtext=None):
00478         def filt(album):
00479             # this filter is more complicated: for each word in the search
00480             # text, it tries to find at least one match on the params of
00481             # the album. If no match is given, then the album doesn't match
00482             if not searchtext:
00483                 return True
00484 
00485             words = RB.search_fold(searchtext).split()
00486             params = list(map(RB.search_fold, [album.name, album.artist,
00487                 album.artists, album.track_titles]))
00488             matches = []
00489 
00490             for word in words:
00491                 match = False
00492 
00493                 for param in params:
00494                     if word in param:
00495                         match = True
00496                         break
00497 
00498                 matches.append(match)
00499 
00500             return False not in matches
00501 
00502         return filt

01206     def _process_queue(self):
01207         '''
01208         Main method that process the albums queue.
01209         First, it tries to adquire a lock on the queue, and if it can, pops
01210         the next element of the queue and process it.
01211         The lock makes sure that only one request is done at a time, and
01212         successfully ignores false timeouts or strand callbacks.
01213         '''
01214         # process the next element in the queue
01215         while self._queue:
01216             album = self._queue.pop(0)
01217 
01218             if album.cover is self.unknown_cover:
01219                 break
01220         else:
01221             album = None
01222 
01223         if album:
01224             # inform the current album being searched
01225             self._callback(album)
01226 
01227             # start the request
01228             self._queue_id += 1
01229             self._search_cover_for_album(album, self._queue_id)
01230 
01231             # add a timeout to the request
01232             Gdk.threads_add_timeout_seconds(GLib.PRIORITY_DEFAULT_IDLE, 40,
01233                     self._next, self._queue_id)
01234         else:
01235             # if there're no more elements, clean the state of the requester
01236             self._running = False
01237             self._callback(None)

01239     def _search_cover_for_album(self, album, search_id):
01240         '''

01242         the callback given once the process finishes (since it generally is
01243         asyncrhonous).
01244         For more information on the callback arguments, check
01245         `RB.ExtDB.request` documentation.
01246 
01247         :param album: `Album` for which search the cover.
01248         '''
01249         # create a key and request the cover
01250         key = album.create_ext_db_key()
01251         provides = self._cover_db.request(key, self._next, search_id)
01252 
01253         if not provides:
01254             # in case there is no provider, call the callback immediately
01255             self._next(search_id)

01260         search_id = args[-1]
01261 
01262         if search_id == self._queue_id:
01263             # only process the next element if the search_id is the same as
01264             # the current id. Otherwise, this is a invalid call
01265             self._process_queue()

01479     def update_cover(self, album, pixbuf=None, uri=None):
01480         '''
01481         Updates the cover database, inserting the pixbuf as the cover art for
01482         all the entries on the album.
01483         In the case a uri is given instead of the pixbuf, it will first try to
01484         retrieve an image from the uri, then recall this method with the
01485         obtained pixbuf.
01486 
01487         :param album: `Album` for which the cover is.
01488         :param pixbuf: `GkdPixbuf.Pixbuf` to use as a cover.
01489         :param uri: `str` from where we should try to retrieve an image.
01490         '''
01491         if pixbuf:
01492             # if it's a pixbuf, assign it to all the artist for the album
01493             key = RB.ExtDBKey.create_storage('album', album.name)
01494             key.add_field('artist', album.artist)
01495 
01496             self._cover_db.store(key, RB.ExtDBSourceType.USER_EXPLICIT,
01497                 pixbuf)
01498 
01499             for artist in album.artists.split(', '):
01500                 key = RB.ExtDBKey.create_storage('album', album.name)
01501                 key.add_field('artist', artist)
01502 
01503                 self._cover_db.store(key, RB.ExtDBSourceType.USER_EXPLICIT,
01504                 pixbuf)
01505 
01506         elif uri:
01507             parsed = rb3compat.urlparse(uri)
01508 
01509             if parsed.scheme == 'file':
01510                 # local file, load it on a pixbuf and assign it
01511                 path = rb3compat.url2pathname(uri.strip()).replace('file://', '')
01512 
01513                 if os.path.exists(path):
01514                     cover = GdkPixbuf.Pixbuf.new_from_file(path)
01515 
01516                     self.update_cover(album, cover)
01517 
01518             else:
01519                 # assume is a remote uri and we have to retrieve the data
01520                 def cover_update(data, album):
01521                     # save the cover on a temp file and open it as a pixbuf
01522                     with tempfile.NamedTemporaryFile(mode='w') as tmp:
01523                         try:
01524                             tmp.write(data)
01525                             tmp.flush()
01526                             cover = GdkPixbuf.Pixbuf.new_from_file(tmp.name)
01527 
01528                             # set the new cover
01529                             self.update_cover(album, cover)
01530                         except:
01531                             print("The URI doesn't point to an image or " + \
01532                                 "the image couldn't be opened.")
01533 
01534                 async = rb.Loader()
01535                 async.get_url(uri, cover_update, album)
