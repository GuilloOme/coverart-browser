00097     def do_search(self, album):
00098         '''
00099         When this method is called, the webview gets refreshed with the info
00100         of the album passed.
00101         '''
00102         if album is self.current_album:
00103             return
00104 
00105         self.current_album = album
00106 
00107         artist = album.artist
00108         album_name = album.name
00109 
00110         if album_name.upper() == "UNKNOWN":
00111             album_name = ""
00112 
00113         if artist.upper() == "UNKNOWN":
00114             artist = ""
00115 
00116         if not(album_name == "" and artist == ""):
00117             artist = rb3compat.unicodestr(artist.replace('&', '&amp;'),
00118                 'utf-8')
00119             album_name = rb3compat.unicodestr(album_name.replace('&', '&amp;'), 'utf-8')
00120             self.render_album_art_search(artist, album_name)
