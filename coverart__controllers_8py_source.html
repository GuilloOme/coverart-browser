00301     def _find_alternates(self, test_genre):
00302         # the following genre checks are required
00303         # 1. if we have user defined genres
00304         # 2. then check locale specific system genres
00305         # 3. then check local specific alternates
00306         # 4. then check if we system genres
00307 
00308         # where necessary check if any of the genres are a substring
00309         # of test_genre - check in reverse order so that we
00310         # test largest strings first (prevents spurious matches with
00311         # short strings)
00312         # N.B. we use RB.search_fold since the strings can be
00313         # in a mixture of cases, both unicode (normalized or not) and str
00314         # and as usual python cannot mix and match these types.
00315 
00316         
00317         test_genre = RB.search_fold(test_genre)
00318         
00319         ret, sprite = self._match_genres(test_genre, self._spritesheet.GENRE_USER)
00320         if ret:
00321             return sprite
00322                 
00323         for genre in sorted(self._spritesheet.locale_names,
00324             key=lambda b: (-len(b), b)):
00325             if RB.search_fold(genre) in test_genre:
00326                 return self._spritesheet[self._spritesheet.locale_names[genre]]
00327 
00328         # next check locale alternates
00329         ret, sprite = self._match_genres(test_genre, self._spritesheet.GENRE_LOCALE)
00330         if ret:
00331             return sprite
00332 
00333         ret, sprite = self._match_genres(test_genre, self._spritesheet.GENRE_SYSTEM)
00334         if ret:
00335             return sprite
00336         
00337         # check if any of the default genres are a substring
00338         # of test_genre - check in reverse order so that we
00339         # test largest strings first (prevents spurious matches with
00340         # short strings)
00341         for genre in sorted(self._spritesheet.names,
00342             key=lambda b: (-len(b), b)):
