00242     def _find_alternates(self, test_genre):
00243         # the following genre checks are required
00244         # 1. if we have locale specific genres check first
00245         # 2. then check locale specific alternates
00246         # 3. then check if we have default genres
00247         # 4. then check if we have default alternates
00248 
00249         
00250         # first check if any of the locale genres are a substring
00251         # of test_genre - check in reverse order so that we
00252         # test largest strings first (prevents spurious matches with
00253         # short strings)
00254         # N.B. we use RB.search_fold since the strings can be
00255         # in a mixture of cases, both unicode (normalized or not) and str
00256         # and as usual python cannot mix and match these types.
00257         
00258         for genre in sorted(self._spritesheet.locale_names,
00259             key=lambda b: (-len(b), b)):
00260             if RB.search_fold(genre) in RB.search_fold(test_genre):
00261                 return self._spritesheet[self._spritesheet.locale_names[genre]]
00262 
00263         # next check locale alternates
00264         case_search = CaseInsensitiveDict(self._spritesheet.locale_alternate)
00265         
00266         if RB.search_fold(test_genre) in case_search:
00267             return self._spritesheet[case_search[RB.search_fold(test_genre)]]
00268 
00269         # check if any of the default genres are a substring
00270         # of test_genre - check in reverse order so that we
00271         # test largest strings first (prevents spurious matches with
00272         # short strings)
00273         for genre in sorted(self._spritesheet.names,
00274             key=lambda b: (-len(b), b)):
00275             if RB.search_fold(genre) in RB.search_fold(test_genre):
00276                 return self._spritesheet[genre]
00277 
00278         # next check alternates
00279         case_search = CaseInsensitiveDict(self._spritesheet.alternate)
00280         if RB.search_fold(test_genre) in case_search:
00281             return self._spritesheet[case_search[RB.search_fold(test_genre)]]
00282 
00283         # if no matches then default to unrecognised image
00284         return self._unrecognised_image
