00129 class SortedCollection(object):
00130     '''Sequence sorted by a key function.
00131 
00132     SortedCollection() is much easier to work with than using bisect() directly.
00133     It supports key functions like those use in sorted(), min(), and max().
00134     The result of the key function call is saved so that keys can be searched
00135     efficiently.
00136 
00137     Instead of returning an insertion-point which can be hard to interpret, the
00138     five find-methods return a specific item in the sequence. They can scan for
00139     exact matches, the last item less-than-or-equal to a key, or the first item
00140     greater-than-or-equal to a key.
00141 
00142     Once found, an item's ordinal position can be located with the index() method.
00143     New items can be added with the insert() and insert_right() methods.
00144     Old items can be deleted with the remove() method.
00145 
00146     The usual sequence methods are provided to support indexing, slicing,
00147     length lookup, clearing, copying, forward and reverse iteration, contains
00148     checking, item counts, item removal, and a nice looking repr.
00149 
00150     Finding and indexing are O(log n) operations while iteration and insertion
00151     are O(n).  The initial sort is O(n log n).
00152 
00153     The key function is stored in the 'key' attibute for easy introspection or
00154     so that you can assign a new key function (triggering an automatic re-sort).
00155 
00156     In short, the class was designed to handle all of the common use cases for
00157     bisect but with a simpler API and support for key functions.
00158 
00159     >>> from pprint import pprint
00160     >>> from operator import itemgetter
00161 
00162     >>> s = SortedCollection(key=itemgetter(2))
00163     >>> for record in [
00164     ...         ('roger', 'young', 30),
00165     ...         ('angela', 'jones', 28),
00166     ...         ('bill', 'smith', 22),
00167     ...         ('david', 'thomas', 32)]:
00168     ...     s.insert(record)
00169 
00170     >>> pprint(list(s))         # show records sorted by age
00171     [('bill', 'smith', 22),
00172      ('angela', 'jones', 28),
00173      ('roger', 'young', 30),
00174      ('david', 'thomas', 32)]
00175 
00176     >>> s.find_le(29)           # find oldest person aged 29 or younger
00177     ('angela', 'jones', 28)
00178     >>> s.find_lt(28)           # find oldest person under 28
00179     ('bill', 'smith', 22)
00180     >>> s.find_gt(28)           # find youngest person over 28
00181     ('roger', 'young', 30)
00182 
00183     >>> r = s.find_ge(32)       # find youngest person aged 32 or older
00184     >>> s.index(r)              # get the index of their record
00185     3
00186     >>> s[3]                    # fetch the record at that index
00187     ('david', 'thomas', 32)
00188 
00189     >>> s.key = itemgetter(0)   # now sort by first name
00190     >>> pprint(list(s))
00191     [('angela', 'jones', 28),
00192      ('bill', 'smith', 22),
00193      ('david', 'thomas', 32),
00194      ('roger', 'young', 30)]
00195 
00196     '''
00197 
00198     def __init__(self, iterable=(), key=None):
00199         self._given_key = key
00200         key = (lambda x: x) if key is None else key
00201         decorated = sorted((key(item), item) for item in iterable)
00202         self._keys = [k for k, item in decorated]
00203         self._items = [item for k, item in decorated]
00204         self._key = key
00205 
00206     def _getkey(self):
00207         return self._key
00208 
00209     def _setkey(self, key):
00210         if key is not self._key:
00211             self.__init__(self._items, key=key)
00212 
00213     def _delkey(self):
00214         self._setkey(None)
00215 
00216     key = property(_getkey, _setkey, _delkey, 'key function')
00217 
00218     def clear(self):
00219         self.__init__([], self._key)
00220 
00221     def copy(self):
00222         return self.__class__(self, self._key)
00223

00225         return len(self._items)
00226 
00227     def __getitem__(self, i):
00228         return self._items[i]
00229 
00230     def __iter__(self):
00231         return iter(self._items)
00232 
00233     def __reversed__(self):
00234         return ReversedSortedCollection(self)

00236     def __repr__(self):
00237         return '%s(%r, key=%s)' % (
00238             self.__class__.__name__,
00239             self._items,
00240             getattr(self._given_key, '__name__', repr(self._given_key))
00241         )

00244         return self.__class__, (self._items, self._given_key)

00246     def __contains__(self, item):
00247         k = self._key(item)
00248         i = bisect_left(self._keys, k)
00249         j = bisect_right(self._keys, k)
00250         return item in self._items[i:j]

00252     def index(self, item):
00253         'Find the position of an item.  Raise ValueError if not found.'
00254         k = self._key(item)
00255         i = bisect_left(self._keys, k)
00256         j = bisect_right(self._keys, k)
00257         return self._items[i:j].index(item) + i
00258 
00259     def count(self, item):
00260         'Return number of occurrences of item'
00261         k = self._key(item)
00262         i = bisect_left(self._keys, k)
00263         j = bisect_right(self._keys, k)
00264         return self._items[i:j].count(item)

00266     def insert(self, item):
00267         'Insert a new item.  If equal keys are found, add to the left'
00268         k = self._key(item)
00269         i = bisect_left(self._keys, k)
00270         self._keys.insert(i, k)
00271         self._items.insert(i, item)
00272 
00273         return i

00275     def reorder(self, item):
00276         '''Reorder an item. If its key changed, then the item is
00277         repositioned, otherwise the item stays untouched'''
00278         index = self._items.index(item)
00279         new_index = -1
00280 
00281         if self._keys[index] != self._key(item):
00282             del self._keys[index]
00283             del self._items[index]
00284 
00285             new_index = self.insert(item)
00286 
00287         return new_index

00289     def insert_all(self, items):
00290         for item in items:
00291             self.insert(item)

00293     def remove(self, item):
00294         'Remove first occurence of item.  Raise ValueError if not found'
00295         i = self.index(item)
00296         del self._keys[i]
00297         del self._items[i]
