00154 class CreateExternalPluginMenu(GObject.Object):
00155     '''
00156     This is the key class called to initialise all supported plugins
00157     
00158     :param section_name: `str` unique name of the section holding the menu items
00159     :param at_position: `int` position within the GtkMenu to create menu - ignored for RB2.99
00160     :param popup: `Menu` whole popupmenu including sub-menus
00161     '''
00162     def __init__(self, section_name, at_position, popup, **kargs):
00163         super(CreateExternalPluginMenu, self).__init__(**kargs)
00164 
00165         self.menu = popup
00166         self.section_name = section_name
00167         self.at_position = at_position
00168         
00169         self._actiongroup = ActionGroup(popup.shell, section_name + '_externalplugins')
00170         
00171         # all supported plugins will be defined in the following array by parsing
00172         # the plugins XML file for the definition.  Supported plugins are split between
00173         # rb2.99 and later and rb2.98 and earlier due to the likelihood that earlier
00174         # plugins may never be updated by their authors
00175         
00176         self.supported_plugins = []
00177         
00178         extplugins = rb.find_plugin_file(popup.plugin, 'ui/coverart_external_plugins.xml')
00179         root = ET.parse(open(extplugins)).getroot()
00180 
00181         if rb3compat.is_rb3(popup.shell):
00182             base = 'rb3/plugin'
00183         else:
00184             base = 'rb2/plugin'
00185 
00186         for elem in root.xpath(base):
00187             pluginname = elem.attrib['name']
00188 
00189             basemenu = base + "[@name='" + pluginname + "']/menu"
00190             
00191             for menuelem in root.xpath(basemenu):
00192                 ext = ExternalPlugin()
00193                 ext.appendattribute('plugin_name', pluginname)
00194 
00195                 label = menuelem.attrib['label']
00196                 if label != "":
00197                     ext.appendattribute('new_menu_name', label)
00198                     baseattrib = basemenu + "[@label='" + label + "']/attribute"
00199                 else:
00200                     baseattrib = basemenu + "/attribute"
00201 
00202                 for attribelem in root.xpath(baseattrib):
00203                     key = attribelem.attrib['name']
00204                     val = attribelem.text
00205                     ext.appendattribute(key, val)
00206 
00207                 self.supported_plugins.append(ext)
00208         
00209     def create_menu(self, menu_name, for_album = False):
00210         '''
00211         method to create the menu items for all supported plugins
00212 
00213         :param menu_name: `str` unique name (GtkMenu) id for the menu to create
00214         :for_album: `bool` - create a menu applicable for Albums
00215           by default a menu is assumed to be applicable to a track in an
00216           EntryView
00217         '''
00218         self.menu_name = menu_name
00219         
00220         self._actiongroup.remove_actions()        
00221         self.menu.remove_menu_items(self.menu_name, self.section_name)
00222         
00223         items_added = False
00224 
00225         for plugin in self.supported_plugins:
00226             new_menu_item = plugin.create_menu_item(self.menu_name, self.section_name,
00227                 self.at_position, self._actiongroup, self.menu, for_album)
00228 
00229             if (not items_added) and new_menu_item:
00230                 items_added = True

00233             self.menu.insert_separator(self.menu_name, self.at_position)
