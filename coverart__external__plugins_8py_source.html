00078     def create_menu_item(self, menubar, section_name, at_position,
00079         save_actiongroup, save_menu, for_album = False):
00080         '''
00081         method to create the menu item appropriate to the plugin.
00082         A plugin can have many menu items - all menuitems are enclosed
00083         in a section.
00084         
00085         :param menubar: `str` name for the GtkMenu - ignored for RB2.99
00086         :param section_name: `str` unique name of the section holding the menu items
00087         :param at_position: `int` position within the GtkMenu to create menu - ignored for RB2.99
00088         :param save_actiongroup: `ActionGroup` container for all menu-item Actions
00089         :param save_menu: `Menu` whole popupmenu including sub-menus
00090         :param for_album: `bool` create the menu for the album - if not given
00091           then its assumed the menu item is appropriate just for tracks
00092         '''
00093         if for_album and not self.attributes['is_album_menu']:
00094             return False
00095             
00096         if not self.is_activated():
00097             return False
00098 
00099         action = ApplicationShell(save_menu.shell).lookup_action(self.attributes['action_group_name'],
00100             self.attributes['action_name'], self.attributes['action_type'])
00101 
00102         if action:
00103             self.attributes['action']=action
00104             
00105             if self.attributes['new_menu_name'] != '':
00106                 self.attributes['label'] = self.attributes['new_menu_name']
00107             else:
00108                 self.attributes['label']=action.label
00109             #self.attributes['sensitive']=action.get_sensitive()
00110         else:
00111             return False
00112 
00113         action = save_actiongroup.add_action(func=self.menuitem_callback,
00114             action_name=self.attributes['action_name'], album=for_album,
00115             shell=save_menu.shell, label=self.attributes['label'])
00116         
00117         new_menu_item = save_menu.insert_menu_item(menubar, section_name,
00118             at_position, action)
00119         return new_menu_item

00153 class CreateExternalPluginMenu(GObject.Object):
00154     '''
00155     This is the key class called to initialise all supported plugins
00156     
00157     :param section_name: `str` unique name of the section holding the menu items
00158     :param at_position: `int` position within the GtkMenu to create menu - ignored for RB2.99
00159     :param popup: `Menu` whole popupmenu including sub-menus
00160     '''
00161     def __init__(self, section_name, at_position, popup, **kargs):
00162         super(CreateExternalPluginMenu, self).__init__(**kargs)
00163 
00164         self.menu = popup
00165         self.section_name = section_name
00166         self.at_position = at_position
00167         
00168         self._actiongroup = ActionGroup(popup.shell, section_name + '_externalplugins')
00169         
00170         # all supported plugins will be defined in the following array by parsing
00171         # the plugins XML file for the definition.  Supported plugins are split between
00172         # rb2.99 and later and rb2.98 and earlier due to the likelihood that earlier
00173         # plugins may never be updated by their authors
00174         
00175         self.supported_plugins = []
00176         
00177         extplugins = rb.find_plugin_file(popup.plugin, 'ui/coverart_external_plugins.xml')
00178         root = ET.parse(open(extplugins)).getroot()
00179 
00180         if rb3compat.is_rb3(popup.shell):
00181             base = 'rb3/plugin'
00182         else:
00183             base = 'rb2/plugin'
00184 
00185         for elem in root.xpath(base):
00186             pluginname = elem.attrib['name']
00187 
00188             basemenu = base + "[@name='" + pluginname + "']/menu"
00189             
00190             for menuelem in root.xpath(basemenu):
00191                 ext = ExternalPlugin()
00192                 ext.appendattribute('plugin_name', pluginname)
00193 
00194                 label = menuelem.attrib['label']
00195                 if label != "":
00196                     ext.appendattribute('new_menu_name', label)
00197                     baseattrib = basemenu + "[@label='" + label + "']/attribute"
00198                 else:
00199                     baseattrib = basemenu + "/attribute"
00200 
00201                 for attribelem in root.xpath(baseattrib):
00202                     key = attribelem.attrib['name']
00203                     val = attribelem.text
00204                     ext.appendattribute(key, val)
00205 
00206                 self.supported_plugins.append(ext)
00207         
00208     def create_menu(self, menu_name, for_album = False):
00209         '''
00210         method to create the menu items for all supported plugins
00211 
00212         :param menu_name: `str` unique name (GtkMenu) id for the menu to create
00213         :for_album: `bool` - create a menu applicable for Albums
00214           by default a menu is assumed to be applicable to a track in an
00215           EntryView
00216         '''
00217         self.menu_name = menu_name
00218         
00219         self._actiongroup.remove_actions()        
00220         self.menu.remove_menu_items(self.menu_name, self.section_name)
00221         
00222         items_added = False
00223 
00224         for plugin in self.supported_plugins:
00225             new_menu_item = plugin.create_menu_item(self.menu_name, self.section_name,
00226                 self.at_position, self._actiongroup, self.menu, for_album)
00227 
00228             if (not items_added) and new_menu_item:
00229                 items_added = True 
00230 
00231         if items_added:
00232             self.menu.insert_separator(self.menu_name, self.at_position)
