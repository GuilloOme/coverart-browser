00056     def _calc_coords(self, points, indent):
00057         coords = []
00058 
00059         from math import cos, pi, sin
00060         step = pi / points
00061 
00062         for i in range(2 * points):
00063             if i % 2:
00064                 x = (sin(step * i) + 1) * 0.5
00065                 y = (cos(step * i) + 1) * 0.5
00066             else:
00067                 x = (sin(step * i) * indent + 1) * 0.5
00068                 y = (cos(step * i) * indent + 1) * 0.5
00069 
00070             coords.append((x, y))
00071         return coords

00073     def layout(self, cr, x, y, w, h):
00074         points = [(sx_sy[0] * w + x, sx_sy[1] * h + y)
00075             for sx_sy in self.coords]
00076         cr.move_to(*points[0])
00077 
00078         for p in points[1:]:
00079             cr.line_to(*p)
00080 
00081         cr.close_path()
